CREATE TABLE user (
    id INT PRIMARY KEY,  
    username VARCHAR(255) SERIAL PRIMARY KEY NOT NULL
    email VARCHAR(255) NOT NULL UNIQUE,  -- Ensures email is unique for each user
    password VARCHAR(255) NOT NULL  -- Store hashed passwords, hence the length
    )
